001package io.dinject.core;
002
003import java.util.HashMap;
004import java.util.List;
005import java.util.Map;
006
007/**
008 * Extended builder that supports supplied beans (mocks) and enriching beans (spy).
009 */
010public class DBuilderExtn extends DBuilder {
011
012  private final Map<Class<?>, EnrichBean> enrichMap = new HashMap<>();
013
014  private final boolean hasSuppliedBeans;
015
016  DBuilderExtn(List<SuppliedBean> suppliedBeans, List<EnrichBean> enrichBeans) {
017    super();
018    this.hasSuppliedBeans = (suppliedBeans != null && !suppliedBeans.isEmpty());
019    if (hasSuppliedBeans) {
020      beanMap.add(suppliedBeans);
021    }
022    if (enrichBeans != null && !enrichBeans.isEmpty()) {
023      for (EnrichBean spy : enrichBeans) {
024        enrichMap.put(spy.getType(), spy);
025      }
026    }
027  }
028
029  /**
030   * Checking for supplied beans (mocks).
031   */
032  @Override
033  public boolean isAddBeanFor(Class<?> addForType, Class<?> injectTarget) {
034    if (hasSuppliedBeans) {
035      return !beanMap.isSupplied(addForType.getName());
036    }
037    return true;
038  }
039
040  /**
041   * Potentially enrich the bean prior to registering with context.
042   */
043  @SuppressWarnings("unchecked")
044  @Override
045  public <T> T enrich(T bean, Class<?>[] types) {
046    EnrichBean<T> enrich = getEnrich(bean, types);
047    return enrich != null ? enrich.enrich(bean) : bean;
048  }
049
050  /**
051   * Search for EnrichBean on the bean or any of the interface types.
052   */
053  @SuppressWarnings("unchecked")
054  private <T> EnrichBean getEnrich(T bean, Class<?>[] types) {
055
056    EnrichBean<T> enrich = enrichMap.get(bean.getClass());
057    if (enrich != null) {
058      return enrich;
059    }
060    for (Class<?> type : types) {
061      enrich = enrichMap.get(type);
062      if (enrich != null) {
063        return enrich;
064      }
065    }
066    return null;
067  }
068
069}
